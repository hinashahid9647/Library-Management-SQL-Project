-- SHOW ALL STUDENTS
SELECT * FROM Students;

-- SHOW ALL BOOKS
SELECT * FROM Books;

-- JOIN: WHO BORROWED WHAT
SELECT 
    Students.student_name,
    Books.title,
    Borrow.borrow_date,
    Borrow.return_date
FROM Borrow
JOIN Students ON Borrow.student_id = Students.student_id
JOIN Books ON Borrow.book_id = Books.book_id;

-- NOT RETURNED BOOKS
SELECT 
    Students.student_name,
    Books.title
FROM Borrow
JOIN Students ON Borrow.student_id = Students.student_id
JOIN Books ON Borrow.book_id = Books.book_id
WHERE return_date IS NULL;

-- TOTAL BORROWED BOOKS
SELECT COUNT(*) AS total_borrowed_books
FROM Borrow;

-- MOST BORROWED BOOK
SELECT 
    Books.title,
    COUNT(*) AS times_borrowed
FROM Borrow
JOIN Books ON Borrow.book_id = Books.book_id
GROUP BY Books.title
ORDER BY times_borrowed DESC;
